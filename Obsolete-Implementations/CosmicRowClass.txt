class CosmicRow(AnnotationRow):
    def __init__(self, symbol, entrezID, diseaseNameSomatic, diseaseNameGermline):
        self.diseaseNameSomatic = diseaseNameSomatic
        self.diseaseNameGermline = diseaseNameGermline
        super(CosmicRow, self).__init__(symbol, entrezID, None, None, None, "Cosmic", None)

    def __eq__(self, other):
        return (isinstance(other, self.__class__) and
                other.symbol == self.symbol and
                other.entrezID == self.entrezID and
                other.diseaseNameSomatic == self.diseaseNameSomatic and
                other.diseaseNameGermline == self.diseaseNameGermline)

    def __hash__(self):
        return hash((self.symbol, self.entrezID, self.diseaseNameSomatic, self.diseaseNameGermline))

    def print(self):
        print(self.symbol, self.entrezID, self.uniprotID, self.ensemblID, self.doid, self.source,
              self.diseaseNameSomatic, self.diseaseNameGermline, sep="\t")
